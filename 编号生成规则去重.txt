	/**
	 * 商品售卖活动，ID=SP+8位数字
	 * 纯说明活动，ID=SM+8位数字
	 * 第三方活动，ID=SF+8位数字
	 * 排行榜活动，ID=PH+8位数字
	 * 内部红包活动，ID=HB+8位数字
	 * 其中八位数字为当前时间戳的后八位数字
	 * @param type:【SP/SM/SF/PH/HB】
	 * @return
	 */
	public String activityNo(String type){
		StringBuffer activityNo = new StringBuffer(type);
		long time = System.currentTimeMillis();
		String timeStr = String.valueOf(time);
		timeStr = timeStr.substring(timeStr.length()-8, timeStr.length());
		activityNo.append(timeStr);
		String activityNoStr = activityNo.toString();
		
		//去重
		ActivityExample activityExample = new ActivityExample();
        activityExample.createCriteria().andActivityNoEqualTo(activityNoStr).andDeleteFlagEqualTo(CommonConstant.NOT_DELETE);
        Activity activity = activityService.selectOneByExample(activityExample);
        if (activity==null) {
			return activityNoStr;
		}else {
			return activityNo(type);
		}
	}
