1、正整数校验
	if (!Pattern.compile("^[1-9]+[0-9]*$").matcher(sellRule.getValue()).find()){
		return AppResultObj.parameterError("活动参与限制,次数只能输入正整数");
	}
	
2、非负数，最多2位小数
	if (!Pattern.compile("^\\d+(\\.\\d{0,2})?$").matcher(goodCard.getPrice().toString()).find()){
		return AppResultObj.parameterError("卡券价格只能输入非负数，最多2位小数；最大数额999999");
	}

非负整数(正整数 + 0) ："^\\d+$" 
正整数："^[0-9]*[1-9][0-9]*$"
