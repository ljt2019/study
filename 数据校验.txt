	private AppResultObj verifyInput(GymInDTO param) {

		AppResultObj appResultObj = null;
		appResultObj = checkImages(param);
		if (appResultObj != null) {
			return appResultObj;
		}
		// 校验场馆名称是否重复
		appResultObj = checkName(param);
		if (appResultObj != null) {
			return appResultObj;
		}
		// 校验开关店时间
		appResultObj = checkTime(param);
		if (appResultObj != null) {
			return appResultObj;
		}
		// 校验成本
		appResultObj = checkCost(param);
		if (appResultObj != null) {
			return appResultObj;
		}
		// 分成角色校验
		appResultObj = checkSharingRole(param);
		if (appResultObj != null) {
			return appResultObj;
		}
		return appResultObj;
	}


三级：
	private AppResultObj checkImages(GymInDTO param) {
		if (CollectionUtils.isEmpty(param.getImagelist())) {
			return AppResultObj.parameterError("图片不能为空！");
		}
		if (param.getImagelist().size() > 5) {
			return AppResultObj.parameterError("图片最多上传5张！");
		}
		for (String str : param.getImagelist()) {
			if(StringUtils.isBlank(str)){
				return AppResultObj.parameterError("图片不能为空！");
			}
		}
		return null;
	}
