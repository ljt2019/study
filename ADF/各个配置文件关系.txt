
====== 各个配置文件之间的关联关系设置 ======

1、在【MATE-INF/adfm.xml】中配置 DataBindings.cpx 文件的存储路径
  <DataBindingRegistry path="com/ly/view/DataBindings.cpx"/>

2、在【DataBindings.cpx】中配置.jsff/.jspx文件的存储路径
  <pageMap>
    <page path="/pages/main.jsff" usageId="com_ly_view_mainPageDef"/>
    <page path="/pages/index.jspx" usageId="com_ly_view_indexPageDef"/>
  </pageMap>
  <pageDefinitionUsages>
    <page id="com_ly_view_mainPageDef" path="pages.mainPageDef"/> 		// pages.mainPageDef 为对应包下的文件名
    <page id="com_ly_view_indexPageDef" path="pages.indexPageDef"/>
  </pageDefinitionUsages>

3、在【xx-taskflow.xml】中配置view：main.jsff 文件存储路径
  <task-flow-definition id="country-taskflow">
    <default-activity id="__1">main</default-activity>
    <view id="main">
      <page>/pages/main.jsff</page>
    </view>
    <use-page-fragments/>
  </task-flow-definition>

4、在【xxPageDef.xml(xx.jspx对应的)】中配置taskflow：引入xx-taskflow.xml中配置的view
  <executables>
    <variableIterator id="variables"/>
    <taskFlow id="countrytaskflow1"
              taskFlowId="/WEB-INF/taskflow/country-taskflow.xml#country-taskflow"
              activation="deferred"
              xmlns="http://xmlns.oracle.com/adf/controller/binding"/>
  </executables>

5、在【xx.jspx】中绑定作用域：
  <f:view>
    <af:document id="d1">
      <af:form id="f1">
        <af:region value="#{bindings.taskflowdefinition1.regionModel}" id="r1"/>
      </af:form>
    </af:document>
  </f:view>
